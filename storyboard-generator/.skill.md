    description: 生成分镜图片
    usage: generate [选项]
    parameters:
      grid:
        type: string
        description: 网格大小 (3x3, 5x5)
        default: 3x3
        enum:
          - 3x3
          - 5x5
      prompts:
        type: string
        description: 分镜提示词，用 | 分隔
        required: true
      style:
        type: string
        description: 图片风格
        default: real
        enum:
          - real
          - anime
          - 3d
          - cartoon
      image:
        type: string
        description: 参考图片路径
        required: false
      output:
        type: string
        description: 输出文件名
        required: false
    examples:
      - generate --grid 3x3 --prompts "镜头1|镜头2|...|镜头9"
      - generate --grid 5x5 --style anime --prompts "镜头1|...|镜头25"
      - generate --grid 3x3 --image ref.jpg --prompts "镜头1：@ref.jpg|...|镜头9"
  cut:
    description: 切割分镜图片
    usage: cut <图片文件> [选项]
    parameters:
      input:
        type: string
        description: 输入图片文件
        required: true
      output:
        type: string
        description: 输出目录
        default: ./cuts
      grid:
        type: string
        description: 网格大小
        default: 3x3
        enum:
          - 3x3
          - 5x5
    examples:
      - cut storyboard.jpg
      - cut storyboard.jpg --output ./cuts
      - cut storyboard.jpg --grid 5x5
  merge:
    description: 合并分镜图片
    usage: merge <图片目录> [选项]
    parameters:
      input:
        type: string
        description: 输入图片目录
        required: true
      grid:
        type: string
        description: 网格大小
        default: 3x3
        enum:
          - 3x3
          - 5x5
      output:
        type: string
        description: 输出文件名
        default: merged.jpg
    examples:
      - merge ./cuts --grid 3x3
      - merge ./cuts --output final.jpg
      - merge ./cuts --grid 5x5 --output final.jpg
